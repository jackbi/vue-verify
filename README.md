# vue-verify-list

> A Vue.js project

## Build Setup

``` bash
# 安装发送验证码插件
```$xslt
npm install vue-verify-list --save
```

# 引用插件
```$xslt
import VueVerifyList from 'vue-verify-list'

Vue.use(VueVerifyList)
```
# 在组件中使用
```$xslt
<vue-verify :time="time" :isSend="isSend" @my-event="goSend"/>

export default {
  data () {
    return {
      time: 0,
      isSend: false
    }
  },
  methods: {
    Timer () {
      if (this.time > 0) {
        this.time = this.time - 1
        setTimeout(this.Timer, 1000)
      }
    },
    goSend () {
      this.time = 60
      this.isSend = true
      this.Timer()
      this.isSend = false
    }
  }
}
```
